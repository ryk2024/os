#include <stdio.h>

struct frame
{
    int pno;
    int counter;

} frames[5];

int ref_str[30];
int np, n, i, j, currtime;

int pagefound(int pno1)
{
    int i;

    for (i = 0; i < n; i++)

        if (frames[i].pno == pno1)
            return i;
    return -1;

} // page_found

int get_lru()
{
    int min = 0;

    for (i = 0; i < n; i++)

        if (frames[i].counter < frames[min].counter)
            min = i;
    return min;
}

int main()
{
    int i, j, pno1, p, page_fault = 0, flag, cnt = 0;
    ;

        scanf("%d", &ref_str[i]);

    for (i = 0; i < n; i++)
    {
        frames[i].pno = -1;
        frames[i].counter = -1;
    }
    printf("\n Page No.\tFrames\t\tPage Fault");
    printf("\n-------------------------------------------------");
    currtime = 1;
    for (p = 0; p < np; p++)
    {
        flag = 0;

            cnt++;
        }
    }
    printf("\n--------------------------------------------------");

    printf("\nNo. of page Hits: %d ", cnt);
}

void make_toks(char *s, char *tok[])
{
    int i = 0;
    char *p;

    p = strtok(s, " ");
    while (p != NULL)
    {
        tok[i++] = p;
        p = strtok(NULL, " ");
    }
    tok[i] = NULL;
}

void count(char *fn, char op)
{
    int fh, cc = 0, wc = 0, lc = 0;
    char c;
    fh = open(fn, O_RDONLY);
    if (fh == -1)
    {
        printf("FIle %s not found. \n", fn);
        return;
    }
    while (read(fh, &c, 1) > 0)
    {
        if (c == ' ')
            wc++;
        else if (c == '\n')
        {
            wc++;
            lc++;
        }
        cc++;
    }
    close(fh);

    switch (op)
    {
    case 'c':
        printf("no. of characters:%d", cc);
        break;
    case 'w':
        printf("no. of words: %d\n", wc);
        break;
    case '|':
        printf("no. of lines: %d\n", lc);
        break;
    }
}
